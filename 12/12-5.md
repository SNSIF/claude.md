# 12-5. 에이전트 앱 만들기

## AI 에이전트란?

AI 에이전트(Agent)는 단순히 질문에 답하는 것을 넘어, 도구를 사용하고 그 결과를 바탕으로 다음 행동을 결정하며 목표를 달성하는 자율적인 AI 시스템입니다.

**에이전트 = Tool Use + 반복 루프**

```
사용자 목표 입력
    ↓
Claude가 계획 수립
    ↓
도구 호출 (파일 읽기, 검색, 계산 등)
    ↓
결과 분석
    ↓
목표 달성? → 완료
아니면     → 다음 도구 호출 (반복)
```

## 기본 에이전트 루프 패턴

```python
import anthropic

client = anthropic.Anthropic()

def run_agent(task: str, tools: list, max_iterations: int = 10):
    """기본 에이전트 루프"""
    messages = [{"role": "user", "content": task}]

    for i in range(max_iterations):
        print(f"[반복 {i + 1}/{max_iterations}]")

        response = client.messages.create(
            model="claude-sonnet-4-6",
            max_tokens=4096,
            tools=tools,
            messages=messages
        )

        # 작업 완료 — Claude가 텍스트로 최종 답변
        if response.stop_reason == "end_turn":
            final_text = next(
                (b.text for b in response.content if b.type == "text"), ""
            )
            return final_text

        # 도구 호출 처리
        if response.stop_reason == "tool_use":
            messages.append({"role": "assistant", "content": response.content})

            tool_results = []
            for block in response.content:
                if block.type == "tool_use":
                    print(f"  → 도구 호출: {block.name}({block.input})")
                    result = execute_tool(block.name, block.input)
                    print(f"  ← 결과: {str(result)[:100]}...")
                    tool_results.append({
                        "type": "tool_result",
                        "tool_use_id": block.id,
                        "content": str(result)
                    })

            messages.append({"role": "user", "content": tool_results})

    return "최대 반복 횟수 초과. 작업을 완료하지 못했습니다."
```

## 실전 에이전트 예제: 파일 분석 에이전트

파일을 읽고, 디렉토리 구조를 파악하고, 코드를 분석하는 에이전트입니다.

```python
import anthropic
import os
import subprocess
import json

client = anthropic.Anthropic()

# 에이전트가 사용할 도구 정의
tools = [
    {
        "name": "read_file",
        "description": "지정된 경로의 파일 내용을 읽습니다",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {
                    "type": "string",
                    "description": "읽을 파일의 경로"
                }
            },
            "required": ["path"]
        }
    },
    {
        "name": "list_directory",
        "description": "디렉토리의 파일 및 폴더 목록을 반환합니다",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {
                    "type": "string",
                    "description": "목록을 볼 디렉토리 경로 (기본값: 현재 디렉토리)"
                }
            }
        }
    },
    {
        "name": "count_lines",
        "description": "파일의 줄 수를 셉니다",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {
                    "type": "string",
                    "description": "줄 수를 셀 파일 경로"
                }
            },
            "required": ["path"]
        }
    }
]

def execute_tool(name: str, inputs: dict):
    """도구를 실제 실행하는 함수"""
    if name == "read_file":
        path = inputs["path"]
        if not os.path.exists(path):
            return f"오류: 파일을 찾을 수 없습니다: {path}"
        with open(path, "r", encoding="utf-8", errors="replace") as f:
            content = f.read()
        # 너무 긴 파일은 잘라냄
        if len(content) > 3000:
            return content[:3000] + "\n... (파일이 길어 일부 생략됨)"
        return content

    elif name == "list_directory":
        path = inputs.get("path", ".")
        if not os.path.exists(path):
            return f"오류: 디렉토리를 찾을 수 없습니다: {path}"
        items = []
        for item in sorted(os.listdir(path)):
            full_path = os.path.join(path, item)
            if os.path.isdir(full_path):
                items.append(f"[DIR]  {item}/")
            else:
                size = os.path.getsize(full_path)
                items.append(f"[FILE] {item} ({size:,} bytes)")
        return "\n".join(items) if items else "(빈 디렉토리)"

    elif name == "count_lines":
        path = inputs["path"]
        if not os.path.exists(path):
            return f"오류: 파일을 찾을 수 없습니다: {path}"
        with open(path, "r", encoding="utf-8", errors="replace") as f:
            lines = f.readlines()
        return f"{len(lines)}줄"

    return f"알 수 없는 도구: {name}"

def run_file_agent(task: str, working_dir: str = ".") -> str:
    """파일 분석 에이전트 실행"""
    system_prompt = f"""당신은 파일 시스템을 분석하는 전문가입니다.
현재 작업 디렉토리: {os.path.abspath(working_dir)}
도구를 사용해 파일을 탐색하고 사용자의 요청을 처리하세요."""

    messages = [{"role": "user", "content": task}]
    max_iterations = 10

    for i in range(max_iterations):
        response = client.messages.create(
            model="claude-sonnet-4-6",
            max_tokens=4096,
            system=system_prompt,
            tools=tools,
            messages=messages
        )

        if response.stop_reason == "end_turn":
            return next((b.text for b in response.content if b.type == "text"), "")

        if response.stop_reason == "tool_use":
            messages.append({"role": "assistant", "content": response.content})
            tool_results = []

            for block in response.content:
                if block.type == "tool_use":
                    result = execute_tool(block.name, block.input)
                    tool_results.append({
                        "type": "tool_result",
                        "tool_use_id": block.id,
                        "content": result
                    })

            messages.append({"role": "user", "content": tool_results})

    return "작업 시간 초과"


# 실행 예시
result = run_file_agent("현재 디렉토리의 Python 파일들을 찾아서 각 파일의 줄 수를 알려줘")
print(result)
```

## Claude Agent SDK 소개

`claude_agent_sdk`는 에이전트 루프를 추상화한 고수준 라이브러리입니다. 반복적인 루프 코드를 작성할 필요 없이 에이전트를 빠르게 구성할 수 있습니다.

```python
from claude_agent_sdk import Agent

agent = Agent(
    model="claude-sonnet-4-6",
    tools=[...],
    system="당신은 코드 분석 전문가입니다."
)

result = agent.run("src/ 폴더의 Python 파일들을 분석해줘")
print(result)
```

### Agent SDK vs 직접 구현 비교

| 항목 | 직접 구현 | Agent SDK |
|------|---------|-----------|
| 코드량 | 많음 | 적음 |
| 제어 수준 | 세밀한 제어 가능 | 추상화됨 |
| 커스터마이징 | 자유로움 | SDK 제약 내에서 |
| 학습 곡선 | 높음 | 낮음 |
| 프로덕션 권장 | 복잡한 요구사항 | 표준적인 에이전트 |

## 비용 관리

에이전트는 여러 번 반복 호출하므로 비용이 빠르게 증가할 수 있습니다.

### max_tokens 제한

```python
response = client.messages.create(
    model="claude-sonnet-4-6",
    max_tokens=2048,  # 너무 높게 설정하지 않음
    ...
)
```

### 루프 횟수 제한

```python
MAX_ITERATIONS = 5  # 무한 루프 방지
for i in range(MAX_ITERATIONS):
    ...
```

### 저렴한 모델 사용 고려

```python
# 간단한 작업에는 Haiku 사용 (약 20배 저렴)
model = "claude-haiku-4-5-20251001"   # 저렴, 빠름
model = "claude-sonnet-4-6"           # 균형
model = "claude-opus-4-6"             # 고성능, 비쌈
```

### 토큰 사용량 모니터링

```python
total_input = 0
total_output = 0

for i in range(max_iterations):
    response = client.messages.create(...)
    total_input += response.usage.input_tokens
    total_output += response.usage.output_tokens

# 1M 토큰 기준 비용 (Sonnet 4.6 기준, 달러)
# 입력: $3 / 1M 토큰
# 출력: $15 / 1M 토큰
cost = (total_input * 3 + total_output * 15) / 1_000_000
print(f"예상 비용: ${cost:.4f}")
```

## 에러 처리 및 재시도 패턴

```python
import anthropic
import time
from anthropic import RateLimitError, APIConnectionError

def robust_agent_call(messages: list, tools: list, retries: int = 3) -> object:
    """재시도 로직이 포함된 안전한 API 호출"""
    for attempt in range(retries):
        try:
            return client.messages.create(
                model="claude-sonnet-4-6",
                max_tokens=4096,
                tools=tools,
                messages=messages
            )
        except RateLimitError:
            wait_time = 60 * (attempt + 1)
            print(f"요청 한도 초과. {wait_time}초 대기...")
            time.sleep(wait_time)
        except APIConnectionError as e:
            wait_time = 2 ** attempt
            print(f"연결 오류 (시도 {attempt + 1}): {e}. {wait_time}초 후 재시도...")
            time.sleep(wait_time)
        except Exception as e:
            print(f"예상치 못한 오류: {e}")
            raise

    raise RuntimeError(f"{retries}회 재시도 후 실패")
```

## 실습: README 자동 생성 에이전트

디렉토리를 분석해서 README.md를 자동 생성하는 에이전트를 만들어보세요.

`readme_agent.py`로 저장하고 실행하세요.

```python
import anthropic
import os

client = anthropic.Anthropic()

tools = [
    {
        "name": "list_directory",
        "description": "디렉토리 파일 목록을 반환합니다",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "디렉토리 경로 (기본: .)"}
            }
        }
    },
    {
        "name": "read_file",
        "description": "파일 내용을 읽습니다 (최대 2000자)",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "파일 경로"}
            },
            "required": ["path"]
        }
    },
    {
        "name": "write_file",
        "description": "파일에 내용을 씁니다",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "파일 경로"},
                "content": {"type": "string", "description": "파일에 쓸 내용"}
            },
            "required": ["path", "content"]
        }
    }
]

def execute_tool(name: str, inputs: dict):
    if name == "list_directory":
        path = inputs.get("path", ".")
        if not os.path.exists(path):
            return f"경로 없음: {path}"
        items = []
        for item in sorted(os.listdir(path)):
            full = os.path.join(path, item)
            prefix = "[DIR] " if os.path.isdir(full) else "[FILE]"
            items.append(f"{prefix} {item}")
        return "\n".join(items) or "(비어 있음)"

    elif name == "read_file":
        path = inputs["path"]
        if not os.path.exists(path):
            return f"파일 없음: {path}"
        with open(path, "r", encoding="utf-8", errors="replace") as f:
            content = f.read(2000)
        return content

    elif name == "write_file":
        path = inputs["path"]
        with open(path, "w", encoding="utf-8") as f:
            f.write(inputs["content"])
        return f"파일 저장 완료: {path}"

    return f"알 수 없는 도구: {name}"

def run_readme_agent(target_dir: str) -> str:
    """README.md 자동 생성 에이전트"""
    task = f"""'{target_dir}' 디렉토리를 분석하고 README.md를 생성해주세요.

작업 순서:
1. 디렉토리 구조 파악 (list_directory 사용)
2. 주요 파일들 내용 확인 (read_file 사용)
3. 프로젝트 성격 파악
4. README.md 작성 및 저장 (write_file 사용)

README.md 포함 내용:
- 프로젝트 이름 및 설명
- 주요 파일/디렉토리 설명
- 설치 및 실행 방법 (추측 가능한 경우)
- 기술 스택 (발견된 것 기반)"""

    messages = [{"role": "user", "content": task}]
    max_iter = 10

    for i in range(max_iter):
        response = client.messages.create(
            model="claude-sonnet-4-6",
            max_tokens=4096,
            tools=tools,
            messages=messages
        )

        if response.stop_reason == "end_turn":
            return next((b.text for b in response.content if b.type == "text"), "완료")

        if response.stop_reason == "tool_use":
            messages.append({"role": "assistant", "content": response.content})
            results = []
            for block in response.content:
                if block.type == "tool_use":
                    print(f"[에이전트] {block.name}: {str(block.input)[:60]}")
                    result = execute_tool(block.name, block.input)
                    results.append({
                        "type": "tool_result",
                        "tool_use_id": block.id,
                        "content": result
                    })
            messages.append({"role": "user", "content": results})

    return "시간 초과"


# 현재 디렉토리에 대한 README 생성
import sys
target = sys.argv[1] if len(sys.argv) > 1 else "."
print(f"'{target}' 디렉토리 분석 중...\n")
result = run_readme_agent(target)
print("\n" + result)
```

**실행 방법:**
```bash
# 현재 디렉토리 분석
python readme_agent.py .

# 특정 디렉토리 분석
python readme_agent.py ./my-project
```
