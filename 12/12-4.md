# 12-4. 구조화된 출력 (Structured Output)

## 구조화된 출력이란?

구조화된 출력(Structured Output)은 Claude의 응답을 JSON 같은 예측 가능한 형식으로 강제하는 기법입니다. 텍스트 응답은 파싱이 어렵지만, 구조화된 출력을 사용하면 애플리케이션에서 Claude의 응답을 직접 데이터로 활용할 수 있습니다.

**활용 사례:**
- 텍스트에서 정보 추출 (이력서, 기사, 리뷰)
- 분류 결과를 코드에서 처리
- 데이터베이스에 바로 저장할 레코드 생성
- API 응답 형식 통일

---

## 방법 1: 시스템 프롬프트로 JSON 요청 (불안정)

가장 단순한 방법이지만 신뢰도가 낮습니다.

```python
import anthropic
import json

client = anthropic.Anthropic()

response = client.messages.create(
    model="claude-sonnet-4-6",
    max_tokens=512,
    system="""반드시 아래 JSON 형식으로만 응답하세요. 다른 텍스트는 포함하지 마세요.
{
  "name": "이름",
  "age": 나이(숫자),
  "occupation": "직업"
}""",
    messages=[{
        "role": "user",
        "content": "홍길동은 30살 개발자입니다."
    }]
)

# 문제: Claude가 ```json ... ``` 블록으로 감쌀 수 있음
# 문제: 필드가 빠지거나 타입이 다를 수 있음
try:
    data = json.loads(response.content[0].text)
except json.JSONDecodeError:
    # 파싱 실패 처리 필요
    data = {}
```

**단점:**
- Claude가 JSON 외에 설명을 추가할 수 있음
- 스키마를 준수하지 않을 수 있음
- 코드에서 별도 파싱 및 검증 로직 필요

## 방법 2: Tool Use 활용 (권장)

Claude가 반드시 정해진 스키마로 응답하도록 강제하는 가장 안정적인 방법입니다.

```python
import anthropic

client = anthropic.Anthropic()

# 추출하고자 하는 데이터 구조를 Tool로 정의
tools = [{
    "name": "extract_person_info",
    "description": "텍스트에서 인물 정보를 추출합니다",
    "input_schema": {
        "type": "object",
        "properties": {
            "name": {
                "type": "string",
                "description": "인물 이름"
            },
            "age": {
                "type": "integer",
                "description": "나이"
            },
            "occupation": {
                "type": "string",
                "description": "직업"
            },
            "skills": {
                "type": "array",
                "items": {"type": "string"},
                "description": "보유 기술 목록"
            }
        },
        "required": ["name"]
    }
}]

response = client.messages.create(
    model="claude-sonnet-4-6",
    max_tokens=1024,
    tools=tools,
    # 반드시 이 도구를 사용하도록 강제
    tool_choice={"type": "tool", "name": "extract_person_info"},
    messages=[{
        "role": "user",
        "content": "홍길동은 30살 개발자로 Python과 JavaScript를 잘 다룹니다."
    }]
)

# tool_use 블록에서 구조화된 데이터 추출
tool_use = next(b for b in response.content if b.type == "tool_use")
data = tool_use.input
print(data)
# {'name': '홍길동', 'age': 30, 'occupation': '개발자', 'skills': ['Python', 'JavaScript']}

# 딕셔너리로 바로 사용 가능
print(f"이름: {data['name']}")
print(f"나이: {data.get('age', '미상')}")
print(f"기술: {', '.join(data.get('skills', []))}")
```

**장점:**
- Claude가 반드시 스키마를 따름
- JSON 파싱 오류 없음
- 타입이 보장됨 (integer, array 등)
- `required` 필드는 반드시 포함

## Claude -p CLI에서 `--output-format json` 플래그

CLI에서 구조화된 출력을 사용하려면 `--output-format json` 플래그를 사용합니다.

```bash
# 기본 JSON 출력
claude -p "서울의 날씨는 어때?" --output-format json

# 출력 예시
{
  "type": "message",
  "content": "서울의 날씨 정보를...",
  "model": "claude-sonnet-4-6",
  "usage": {
    "input_tokens": 15,
    "output_tokens": 82
  }
}
```

파이프라인에서 유용합니다.

```bash
# jq로 응답 텍스트만 추출
claude -p "오늘 할 일 3가지 추천해줘" --output-format json | jq -r '.content'
```

## Pydantic 모델 통합 (Python)

Pydantic을 사용하면 데이터 검증과 타입 변환을 자동으로 처리할 수 있습니다.

```python
import anthropic
from pydantic import BaseModel, Field
from typing import Optional

client = anthropic.Anthropic()

# Pydantic 모델 정의
class PersonInfo(BaseModel):
    name: str
    age: Optional[int] = None
    occupation: Optional[str] = None
    skills: list[str] = Field(default_factory=list)
    email: Optional[str] = None

# Pydantic 모델에서 Tool 스키마 자동 생성
def pydantic_to_tool(name: str, description: str, model: type[BaseModel]) -> dict:
    schema = model.model_json_schema()
    # Pydantic의 $defs 같은 메타데이터 제거
    schema.pop("title", None)
    return {
        "name": name,
        "description": description,
        "input_schema": schema
    }

tools = [pydantic_to_tool(
    "extract_person",
    "텍스트에서 인물 정보를 추출합니다",
    PersonInfo
)]

def extract_person(text: str) -> PersonInfo:
    response = client.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=512,
        tools=tools,
        tool_choice={"type": "tool", "name": "extract_person"},
        messages=[{"role": "user", "content": text}]
    )
    tool_use = next(b for b in response.content if b.type == "tool_use")
    # Pydantic 모델로 자동 검증 및 변환
    return PersonInfo(**tool_use.input)


# 사용
person = extract_person(
    "김민지는 25살 디자이너로, Figma와 Adobe XD를 사용합니다. 이메일: minji@example.com"
)
print(person.name)       # 김민지
print(person.age)        # 25
print(person.skills)     # ['Figma', 'Adobe XD']
print(person.email)      # minji@example.com
print(person.model_dump())  # 딕셔너리로 변환
```

## Zod 스키마 통합 (TypeScript)

TypeScript에서는 Zod를 활용해 타입 안전성을 보장할 수 있습니다.

```typescript
import Anthropic from "@anthropic-ai/sdk";
import { z } from "zod";

const client = new Anthropic();

// Zod 스키마 정의
const PersonInfoSchema = z.object({
    name: z.string().describe("인물 이름"),
    age: z.number().int().optional().describe("나이"),
    occupation: z.string().optional().describe("직업"),
    skills: z.array(z.string()).default([]).describe("보유 기술 목록"),
    email: z.string().email().optional().describe("이메일 주소")
});

type PersonInfo = z.infer<typeof PersonInfoSchema>;

const tools: Anthropic.Tool[] = [
    {
        name: "extract_person",
        description: "텍스트에서 인물 정보를 추출합니다",
        input_schema: {
            type: "object",
            properties: {
                name: { type: "string", description: "인물 이름" },
                age: { type: "number", description: "나이" },
                occupation: { type: "string", description: "직업" },
                skills: {
                    type: "array",
                    items: { type: "string" },
                    description: "보유 기술 목록"
                },
                email: { type: "string", description: "이메일 주소" }
            },
            required: ["name"]
        }
    }
];

async function extractPerson(text: string): Promise<PersonInfo> {
    const response = await client.messages.create({
        model: "claude-sonnet-4-6",
        max_tokens: 512,
        tools,
        tool_choice: { type: "tool", name: "extract_person" },
        messages: [{ role: "user", content: text }]
    });

    const toolUse = response.content.find(b => b.type === "tool_use");
    if (!toolUse || toolUse.type !== "tool_use") {
        throw new Error("도구 호출 결과가 없습니다");
    }

    // Zod로 타입 검증
    return PersonInfoSchema.parse(toolUse.input);
}

const person = await extractPerson(
    "이준호는 28살 백엔드 개발자로 Go와 Kubernetes를 다룹니다."
);
console.log(person.name);    // 이준호
console.log(person.skills);  // ['Go', 'Kubernetes']
```

## 실습: 이력서 텍스트에서 구조화된 정보 추출

`resume_extractor.py`로 저장하고 실행해보세요.

```python
import anthropic
import json

client = anthropic.Anthropic()

# 이력서 정보 스키마 정의
tools = [{
    "name": "extract_resume",
    "description": "이력서 텍스트에서 구조화된 정보를 추출합니다",
    "input_schema": {
        "type": "object",
        "properties": {
            "name": {
                "type": "string",
                "description": "지원자 이름"
            },
            "email": {
                "type": "string",
                "description": "이메일 주소"
            },
            "phone": {
                "type": "string",
                "description": "연락처"
            },
            "years_of_experience": {
                "type": "number",
                "description": "총 경력 연수"
            },
            "skills": {
                "type": "array",
                "items": {"type": "string"},
                "description": "기술 스택 목록"
            },
            "education": {
                "type": "object",
                "properties": {
                    "degree": {"type": "string", "description": "학위"},
                    "major": {"type": "string", "description": "전공"},
                    "school": {"type": "string", "description": "학교명"}
                },
                "description": "최종 학력"
            },
            "previous_companies": {
                "type": "array",
                "items": {"type": "string"},
                "description": "이전 근무 회사 목록"
            },
            "summary": {
                "type": "string",
                "description": "지원자 역량 요약 (한 문장)"
            }
        },
        "required": ["name", "skills"]
    }
}]

def extract_resume_info(resume_text: str) -> dict:
    response = client.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=1024,
        tools=tools,
        tool_choice={"type": "tool", "name": "extract_resume"},
        messages=[{
            "role": "user",
            "content": f"다음 이력서에서 정보를 추출해주세요:\n\n{resume_text}"
        }]
    )
    tool_use = next(b for b in response.content if b.type == "tool_use")
    return tool_use.input


# 테스트 이력서
sample_resume = """
이름: 박서연
연락처: 010-1234-5678 | seo@email.com

[경력] 5년
- 네이버 (2022~현재): 시니어 프론트엔드 개발자
- 카카오 (2019~2022): 프론트엔드 개발자

[기술]
React, TypeScript, Next.js, GraphQL, Node.js, AWS

[학력]
서울대학교 컴퓨터공학과 학사 (2019)

뛰어난 UI/UX 구현 능력과 팀 협업 경험을 보유한 프론트엔드 전문가.
"""

result = extract_resume_info(sample_resume)

print("=== 추출된 이력서 정보 ===")
print(json.dumps(result, ensure_ascii=False, indent=2))
print()
print(f"이름: {result['name']}")
print(f"기술: {', '.join(result['skills'])}")
print(f"경력: {result.get('years_of_experience', '?')}년")
print(f"요약: {result.get('summary', '')}")
```

**출력 예시:**
```json
{
  "name": "박서연",
  "email": "seo@email.com",
  "phone": "010-1234-5678",
  "years_of_experience": 5,
  "skills": ["React", "TypeScript", "Next.js", "GraphQL", "Node.js", "AWS"],
  "education": {
    "degree": "학사",
    "major": "컴퓨터공학",
    "school": "서울대학교"
  },
  "previous_companies": ["네이버", "카카오"],
  "summary": "5년 경력의 프론트엔드 전문가로 React/TypeScript 기반 UI 개발에 강점을 보유"
}
```
