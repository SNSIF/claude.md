# 12-2. Tool Use (함수 호출)

## Tool Use란?

Tool Use(도구 사용)는 Claude에게 외부 함수나 API를 호출하는 능력을 부여하는 기능입니다. Claude 자체는 실시간 데이터에 접근하거나 코드를 직접 실행할 수 없지만, Tool Use를 통해 개발자가 정의한 함수를 Claude가 적절한 타이밍에 호출하도록 지시할 수 있습니다.

**활용 예시:**
- 날씨 API 호출 → 현재 날씨 정보 제공
- 데이터베이스 조회 → 실시간 상품/재고 정보
- 계산 수행 → 복잡한 수치 계산
- 파일 읽기/쓰기 → 로컬 시스템 작업
- 검색 엔진 쿼리 → 최신 정보 검색

---

## Tool 정의 방법

도구는 JSON Schema 형식으로 정의합니다. Claude는 이 정의를 보고 언제 어떻게 도구를 호출할지 판단합니다.

```python
tools = [
    {
        "name": "get_weather",
        "description": "특정 도시의 현재 날씨를 가져옵니다",
        "input_schema": {
            "type": "object",
            "properties": {
                "city": {
                    "type": "string",
                    "description": "날씨를 조회할 도시 이름 (영문)"
                },
                "unit": {
                    "type": "string",
                    "enum": ["celsius", "fahrenheit"],
                    "description": "온도 단위"
                }
            },
            "required": ["city"]
        }
    }
]
```

**좋은 description 작성 요령:**
- 도구가 무엇을 하는지 명확하게 설명
- 언제 사용해야 하는지 포함
- 반환값의 형태도 명시하면 좋음

---

## Tool 호출 전체 흐름

Tool Use는 단순한 요청-응답이 아닌, 여러 단계를 거치는 대화입니다.

```
1. 사용자 메시지 전송
        ↓
2. Claude가 tool_use 블록으로 응답
   (어떤 도구를, 어떤 인자로 호출할지 결정)
        ↓
3. 앱에서 실제 함수 실행
        ↓
4. tool_result를 messages에 추가
        ↓
5. Claude가 결과를 바탕으로 최종 응답 생성
```

---

## 완전한 Python 예제 — 날씨 봇

```python
import anthropic
import json

client = anthropic.Anthropic()

# 도구 정의
tools = [
    {
        "name": "get_weather",
        "description": "특정 도시의 현재 날씨 정보를 반환합니다. 기온, 날씨 상태, 습도를 포함합니다.",
        "input_schema": {
            "type": "object",
            "properties": {
                "city": {
                    "type": "string",
                    "description": "날씨를 조회할 도시 이름 (영문, 예: Seoul, Tokyo, New York)"
                },
                "unit": {
                    "type": "string",
                    "enum": ["celsius", "fahrenheit"],
                    "description": "온도 단위. 기본값: celsius"
                }
            },
            "required": ["city"]
        }
    }
]

# 실제 날씨 API 대신 mock 함수 사용
def get_weather(city: str, unit: str = "celsius") -> dict:
    # 실제 구현에서는 OpenWeatherMap 등의 API 호출
    mock_data = {
        "Seoul":     {"temp": 18, "condition": "맑음", "humidity": 60},
        "Tokyo":     {"temp": 22, "condition": "흐림", "humidity": 70},
        "New York":  {"temp": 15, "condition": "비",   "humidity": 85},
    }
    data = mock_data.get(city, {"temp": 20, "condition": "알 수 없음", "humidity": 50})
    temp = data["temp"]
    if unit == "fahrenheit":
        temp = temp * 9 / 5 + 32
    return {
        "city": city,
        "temperature": f"{temp}°{'C' if unit == 'celsius' else 'F'}",
        "condition": data["condition"],
        "humidity": f"{data['humidity']}%"
    }

def execute_tool(name: str, inputs: dict):
    """도구 이름에 따라 실제 함수 실행"""
    if name == "get_weather":
        return get_weather(**inputs)
    raise ValueError(f"알 수 없는 도구: {name}")

def weather_bot(user_message: str) -> str:
    messages = [{"role": "user", "content": user_message}]

    # 1단계: Claude에게 요청 (Claude가 도구 호출 여부를 결정)
    response = client.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=1024,
        tools=tools,
        messages=messages
    )

    # stop_reason이 "tool_use"이면 Claude가 도구를 호출하길 원하는 것
    while response.stop_reason == "tool_use":
        # 2단계: Claude의 응답(tool_use 블록 포함)을 messages에 추가
        messages.append({"role": "assistant", "content": response.content})

        # 3단계: 각 도구 호출을 실행하고 결과 수집
        tool_results = []
        for block in response.content:
            if block.type == "tool_use":
                print(f"[도구 호출] {block.name}({block.input})")
                result = execute_tool(block.name, block.input)
                print(f"[도구 결과] {result}")

                tool_results.append({
                    "type": "tool_result",
                    "tool_use_id": block.id,
                    "content": json.dumps(result, ensure_ascii=False)
                })

        # 4단계: 도구 결과를 messages에 추가
        messages.append({"role": "user", "content": tool_results})

        # 5단계: Claude에게 다시 요청 (최종 응답 생성)
        response = client.messages.create(
            model="claude-sonnet-4-6",
            max_tokens=1024,
            tools=tools,
            messages=messages
        )

    # 최종 텍스트 응답 반환
    return response.content[0].text


# 테스트
print(weather_bot("서울과 도쿄의 날씨를 알려줘"))
```

**출력 예시:**
```
[도구 호출] get_weather({'city': 'Seoul', 'unit': 'celsius'})
[도구 결과] {'city': 'Seoul', 'temperature': '18°C', 'condition': '맑음', 'humidity': '60%'}
[도구 호출] get_weather({'city': 'Tokyo', 'unit': 'celsius'})
[도구 결과] {'city': 'Tokyo', 'temperature': '22°C', 'condition': '흐림', 'humidity': '70%'}

현재 날씨 정보입니다:
- 서울: 18°C, 맑음, 습도 60%
- 도쿄: 22°C, 흐림, 습도 70%

도쿄가 서울보다 4도 더 따뜻하고, 습도는 도쿄가 더 높네요!
```

## TypeScript 예제

```typescript
import Anthropic from "@anthropic-ai/sdk";

const client = new Anthropic();

const tools: Anthropic.Tool[] = [
    {
        name: "get_weather",
        description: "특정 도시의 현재 날씨를 반환합니다",
        input_schema: {
            type: "object",
            properties: {
                city: {
                    type: "string",
                    description: "도시 이름 (영문)"
                },
                unit: {
                    type: "string",
                    enum: ["celsius", "fahrenheit"],
                    description: "온도 단위"
                }
            },
            required: ["city"]
        }
    }
];

function getWeather(city: string, unit = "celsius"): object {
    // mock 구현
    return { city, temperature: `20°C`, condition: "맑음", humidity: "60%" };
}

async function weatherBot(userMessage: string): Promise<string> {
    const messages: Anthropic.MessageParam[] = [
        { role: "user", content: userMessage }
    ];

    let response = await client.messages.create({
        model: "claude-sonnet-4-6",
        max_tokens: 1024,
        tools,
        messages
    });

    while (response.stop_reason === "tool_use") {
        messages.push({ role: "assistant", content: response.content });

        const toolResults: Anthropic.ToolResultBlockParam[] = [];

        for (const block of response.content) {
            if (block.type === "tool_use") {
                const input = block.input as { city: string; unit?: string };
                const result = getWeather(input.city, input.unit);
                toolResults.push({
                    type: "tool_result",
                    tool_use_id: block.id,
                    content: JSON.stringify(result)
                });
            }
        }

        messages.push({ role: "user", content: toolResults });

        response = await client.messages.create({
            model: "claude-sonnet-4-6",
            max_tokens: 1024,
            tools,
            messages
        });
    }

    const textBlock = response.content.find(b => b.type === "text");
    return textBlock && textBlock.type === "text" ? textBlock.text : "";
}

weatherBot("뉴욕 날씨 어때?").then(console.log);
```

## 병렬 Tool Use

Claude는 여러 도구를 동시에(한 번의 응답에서) 호출할 수 있습니다.

```python
# Claude가 서울, 도쿄, 뉴욕을 한 번에 조회할 수 있음
response = client.messages.create(
    model="claude-sonnet-4-6",
    max_tokens=1024,
    tools=tools,
    messages=[{"role": "user", "content": "서울, 도쿄, 뉴욕 날씨를 모두 알려줘"}]
)

# response.content에 여러 tool_use 블록이 포함됨
for block in response.content:
    if block.type == "tool_use":
        print(f"{block.name}: {block.input}")
# get_weather: {'city': 'Seoul'}
# get_weather: {'city': 'Tokyo'}
# get_weather: {'city': 'New York'}
```

병렬 도구 호출을 처리할 때는 모든 `tool_result`를 하나의 user 메시지에 담아야 합니다.

## 강제 도구 호출

Claude가 반드시 특정 도구를 사용하도록 강제할 수 있습니다.

```python
response = client.messages.create(
    model="claude-sonnet-4-6",
    max_tokens=1024,
    tools=tools,
    # 반드시 get_weather 도구를 사용하도록 강제
    tool_choice={"type": "tool", "name": "get_weather"},
    messages=[{"role": "user", "content": "오늘 날씨가 좋을까?"}]
)
```

| `tool_choice` 옵션 | 의미 |
|-------------------|------|
| `{"type": "auto"}` | Claude가 도구 사용 여부를 자율적으로 결정 (기본값) |
| `{"type": "any"}` | 반드시 도구 중 하나를 사용 (어떤 것이든) |
| `{"type": "tool", "name": "..."}` | 반드시 지정한 도구를 사용 |
| `{"type": "none"}` | 도구 사용 금지 |

## 실습: 계산기 도구 만들기

`calculator_bot.py`로 저장하고 실행해보세요.

```python
import anthropic
import json

client = anthropic.Anthropic()

# 계산기 도구 정의
tools = [
    {
        "name": "calculate",
        "description": "두 숫자 사이의 사칙연산을 수행합니다. 수학 계산이 필요한 모든 경우에 사용하세요.",
        "input_schema": {
            "type": "object",
            "properties": {
                "operation": {
                    "type": "string",
                    "enum": ["add", "subtract", "multiply", "divide"],
                    "description": "수행할 연산: add(덧셈), subtract(뺄셈), multiply(곱셈), divide(나눗셈)"
                },
                "a": {
                    "type": "number",
                    "description": "첫 번째 피연산자"
                },
                "b": {
                    "type": "number",
                    "description": "두 번째 피연산자"
                }
            },
            "required": ["operation", "a", "b"]
        }
    }
]

def calculate(operation: str, a: float, b: float) -> dict:
    """실제 계산 수행"""
    if operation == "add":
        result = a + b
    elif operation == "subtract":
        result = a - b
    elif operation == "multiply":
        result = a * b
    elif operation == "divide":
        if b == 0:
            return {"error": "0으로 나눌 수 없습니다"}
        result = a / b
    else:
        return {"error": f"알 수 없는 연산: {operation}"}
    return {"operation": operation, "a": a, "b": b, "result": result}

def calculator_bot(user_message: str) -> str:
    messages = [{"role": "user", "content": user_message}]

    response = client.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=1024,
        tools=tools,
        messages=messages
    )

    while response.stop_reason == "tool_use":
        messages.append({"role": "assistant", "content": response.content})
        tool_results = []

        for block in response.content:
            if block.type == "tool_use":
                result = calculate(**block.input)
                tool_results.append({
                    "type": "tool_result",
                    "tool_use_id": block.id,
                    "content": json.dumps(result)
                })

        messages.append({"role": "user", "content": tool_results})
        response = client.messages.create(
            model="claude-sonnet-4-6",
            max_tokens=1024,
            tools=tools,
            messages=messages
        )

    return response.content[0].text


# 테스트
print(calculator_bot("123 곱하기 456은 얼마야?"))
print(calculator_bot("1000에서 375를 빼고, 그 결과를 7로 나눠줘"))
```
